## SAMPLE QUERY PATTERNS

```sql
-- Total active users in October
SELECT COUNT(DISTINCT user_id) as active_users
FROM user_activity
WHERE activity_date BETWEEN '2025-10-01' AND '2025-10-31';

-- Revenue by product line
SELECT
  product_line,
  SUM(revenue) as total_revenue,
  COUNT(DISTINCT user_id) as active_users,
  SUM(revenue) / COUNT(DISTINCT user_id) as revenue_per_user
FROM revenue_events
WHERE event_date BETWEEN '2025-10-01' AND '2025-10-31'
GROUP BY product_line;

-- MoM growth
WITH oct_metrics AS (
  SELECT COUNT(DISTINCT user_id) as users
  FROM user_activity
  WHERE activity_date BETWEEN '2025-10-01' AND '2025-10-31'
),
sept_metrics AS (
  SELECT COUNT(DISTINCT user_id) as users
  FROM user_activity
  WHERE activity_date BETWEEN '2025-09-01' AND '2025-09-30'
)
SELECT
  o.users as oct_users,
  s.users as sept_users,
  ROUND(((o.users - s.users) * 100.0 / s.users), 2) as growth_pct
FROM oct_metrics o, sept_metrics s;

-- User segmentation by activity
SELECT
  CASE
    WHEN transaction_count >= 20 THEN 'Power User'
    WHEN transaction_count >= 5 THEN 'Regular User'
    WHEN transaction_count >= 1 THEN 'Occasional User'
    ELSE 'Dormant User'
  END as user_segment,
  COUNT(*) as user_count,
  AVG(transaction_value) as avg_transaction_value
FROM (
  SELECT
    user_id,
    COUNT(*) as transaction_count,
    SUM(amount) as transaction_value
  FROM transactions
  WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
  GROUP BY user_id
)
GROUP BY user_segment
ORDER BY user_count DESC;
```
