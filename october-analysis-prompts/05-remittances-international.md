## 10. SAMPLE SQL QUERIES

```sql
-- Total remittance volume and count
SELECT
  COUNT(*) as total_remittances,
  SUM(amount) as total_volume_usd,
  AVG(amount) as avg_amount,
  PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY amount) as median_amount
FROM remittances
WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'COMPLETED';

-- Active remittance users
SELECT
  COUNT(DISTINCT user_id) as active_remittance_users,
  AVG(user_tx_count) as avg_remittances_per_user
FROM (
  SELECT user_id, COUNT(*) as user_tx_count
  FROM remittances
  WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
    AND status = 'COMPLETED'
  GROUP BY user_id
);

-- Top destination countries
SELECT
  destination_country,
  COUNT(*) as transaction_count,
  SUM(amount) as total_volume,
  AVG(amount) as avg_amount,
  ROUND(SUM(amount) * 100.0 / (SELECT SUM(amount) FROM remittances WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31' AND status = 'COMPLETED'), 2) as pct_of_total_volume
FROM remittances
WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'COMPLETED'
GROUP BY destination_country
ORDER BY total_volume DESC
LIMIT 10;

-- Delivery method breakdown
SELECT
  delivery_method,
  COUNT(*) as count,
  AVG(TIMESTAMPDIFF(HOUR, transaction_date, delivery_date)) as avg_delivery_hours
FROM remittances
WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'COMPLETED'
GROUP BY delivery_method
ORDER BY count DESC;

-- Fee analysis
SELECT
  SUM(fee) as total_fee_revenue,
  AVG(fee) as avg_fee,
  AVG(fee / amount * 100) as avg_fee_pct
FROM remittances
WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'COMPLETED';

-- Failed remittances
SELECT
  failure_reason,
  COUNT(*) as failure_count,
  ROUND(COUNT(*) * 100.0 / (SELECT COUNT(*) FROM remittances WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'), 2) as failure_rate_pct
FROM remittances
WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'FAILED'
GROUP BY failure_reason
ORDER BY failure_count DESC;

-- User segmentation
SELECT
  CASE
    WHEN tx_count >= 6 THEN 'Power Sender'
    WHEN tx_count >= 2 THEN 'Regular Sender'
    ELSE 'Occasional Sender'
  END as user_segment,
  COUNT(*) as user_count,
  AVG(tx_count) as avg_remittances,
  AVG(total_amount) as avg_volume
FROM (
  SELECT
    user_id,
    COUNT(*) as tx_count,
    SUM(amount) as total_amount
  FROM remittances
  WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
    AND status = 'COMPLETED'
  GROUP BY user_id
)
GROUP BY user_segment
ORDER BY avg_remittances DESC;
```
