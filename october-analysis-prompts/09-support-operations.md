## 11. SAMPLE SQL QUERIES

```sql
-- Total support tickets in October
SELECT
  COUNT(*) as total_tickets,
  COUNT(CASE WHEN status = 'RESOLVED' THEN 1 END) as resolved_tickets,
  ROUND(COUNT(CASE WHEN status = 'RESOLVED' THEN 1 END) * 100.0 / COUNT(*), 2) as resolution_rate_pct
FROM support_tickets
WHERE created_at BETWEEN '2025-10-01' AND '2025-10-31';

-- Top issue categories
SELECT
  issue_category,
  COUNT(*) as ticket_count,
  ROUND(COUNT(*) * 100.0 / (SELECT COUNT(*) FROM support_tickets WHERE created_at BETWEEN '2025-10-01' AND '2025-10-31'), 2) as pct_of_total,
  AVG(TIMESTAMPDIFF(HOUR, created_at, resolved_at)) as avg_resolution_hours
FROM support_tickets
WHERE created_at BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'RESOLVED'
GROUP BY issue_category
ORDER BY ticket_count DESC
LIMIT 10;

-- Support channel breakdown
SELECT
  support_channel,
  COUNT(*) as ticket_count,
  AVG(TIMESTAMPDIFF(HOUR, created_at, resolved_at)) as avg_resolution_hours,
  ROUND(COUNT(CASE WHEN first_contact_resolution = TRUE THEN 1 END) * 100.0 / COUNT(*), 2) as first_contact_resolution_pct
FROM support_tickets
WHERE created_at BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'RESOLVED'
GROUP BY support_channel
ORDER BY ticket_count DESC;

-- Resolution time distribution
SELECT
  CASE
    WHEN TIMESTAMPDIFF(HOUR, created_at, resolved_at) < 1 THEN '<1 hour'
    WHEN TIMESTAMPDIFF(HOUR, created_at, resolved_at) < 4 THEN '1-4 hours'
    WHEN TIMESTAMPDIFF(HOUR, created_at, resolved_at) < 12 THEN '4-12 hours'
    WHEN TIMESTAMPDIFF(HOUR, created_at, resolved_at) < 24 THEN '12-24 hours'
    WHEN TIMESTAMPDIFF(HOUR, created_at, resolved_at) < 72 THEN '1-3 days'
    ELSE '3+ days'
  END as resolution_time_bucket,
  COUNT(*) as ticket_count,
  ROUND(COUNT(*) * 100.0 / (SELECT COUNT(*) FROM support_tickets WHERE created_at BETWEEN '2025-10-01' AND '2025-10-31' AND status = 'RESOLVED'), 2) as pct
FROM support_tickets
WHERE created_at BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'RESOLVED'
GROUP BY resolution_time_bucket
ORDER BY MIN(TIMESTAMPDIFF(HOUR, created_at, resolved_at));

-- Platform uptime
SELECT
  ROUND((SUM(uptime_minutes) / (31 * 24 * 60)) * 100, 3) as uptime_pct,
  COUNT(*) as total_incidents,
  SUM(downtime_minutes) as total_downtime_minutes
FROM system_availability
WHERE date BETWEEN '2025-10-01' AND '2025-10-31';

-- TabaPay performance
SELECT
  COUNT(*) as total_tabapay_requests,
  COUNT(CASE WHEN status = 'SUCCESS' THEN 1 END) as successful_requests,
  ROUND(COUNT(CASE WHEN status = 'SUCCESS' THEN 1 END) * 100.0 / COUNT(*), 2) as success_rate_pct,
  AVG(response_time_ms) as avg_response_time_ms
FROM tabapay_api_logs
WHERE request_date BETWEEN '2025-10-01' AND '2025-10-31';
```
