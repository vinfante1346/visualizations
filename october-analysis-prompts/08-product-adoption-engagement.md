## 9. SAMPLE SQL QUERIES

```sql
-- Multi-product usage breakdown
WITH user_products AS (
  SELECT
    user_id,
    (CASE WHEN has_active_card = TRUE THEN 1 ELSE 0 END +
     CASE WHEN has_bambupay = TRUE THEN 1 ELSE 0 END +
     CASE WHEN has_direct_deposit = TRUE THEN 1 ELSE 0 END +
     CASE WHEN has_used_remittances = TRUE THEN 1 ELSE 0 END) as product_count
  FROM user_product_usage
  WHERE activity_month = '2025-10'
)
SELECT
  CASE
    WHEN product_count = 1 THEN 'Single product'
    WHEN product_count = 2 THEN 'Two products'
    WHEN product_count >= 3 THEN 'Three+ products'
    ELSE 'No products'
  END as user_category,
  COUNT(*) as user_count,
  ROUND(COUNT(*) * 100.0 / (SELECT COUNT(*) FROM user_products), 2) as pct_of_users
FROM user_products
GROUP BY user_category
ORDER BY product_count;

-- Product penetration rates
WITH total_users AS (
  SELECT COUNT(DISTINCT user_id) as total
  FROM users
  WHERE status = 'ACTIVE'
)
SELECT
  'Debit card' as product,
  COUNT(DISTINCT user_id) as active_users,
  ROUND(COUNT(DISTINCT user_id) * 100.0 / (SELECT total FROM total_users), 2) as penetration_rate_pct
FROM user_product_usage
WHERE has_active_card = TRUE
  AND activity_month = '2025-10'
UNION ALL
SELECT
  'BambuPay active',
  COUNT(DISTINCT user_id),
  ROUND(COUNT(DISTINCT user_id) * 100.0 / (SELECT total FROM total_users), 2)
FROM bambupay_transactions
WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'COMPLETED';

-- App session metrics
SELECT
  COUNT(*) as total_sessions,
  COUNT(DISTINCT user_id) as active_users,
  ROUND(COUNT(*) / COUNT(DISTINCT user_id), 2) as avg_sessions_per_user,
  AVG(session_duration_seconds) as avg_session_duration_sec
FROM app_sessions
WHERE session_date BETWEEN '2025-10-01' AND '2025-10-31';

-- User segmentation by activity level
WITH user_activity AS (
  SELECT
    user_id,
    COUNT(*) as transaction_count,
    SUM(amount) as total_volume
  FROM transactions
  WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
    AND status = 'COMPLETED'
  GROUP BY user_id
),
percentiles AS (
  SELECT PERCENTILE_CONT(0.9) WITHIN GROUP (ORDER BY total_volume) as p90_volume
  FROM user_activity
)
SELECT
  CASE
    WHEN a.total_volume >= p.p90_volume THEN 'Power User'
    WHEN a.transaction_count >= 10 THEN 'Regular User'
    WHEN a.transaction_count >= 1 THEN 'Occasional User'
    ELSE 'Inactive'
  END as user_segment,
  COUNT(*) as user_count,
  AVG(a.transaction_count) as avg_transactions,
  AVG(a.total_volume) as avg_volume
FROM user_activity a, percentiles p
GROUP BY user_segment
ORDER BY avg_volume DESC;

-- Day of week transaction patterns
SELECT
  DAYNAME(transaction_date) as day_of_week,
  COUNT(*) as transaction_count,
  ROUND(COUNT(*) * 100.0 / (SELECT COUNT(*) FROM transactions WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'), 2) as pct_of_weekly_total,
  AVG(amount) as avg_transaction_value
FROM transactions
WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'COMPLETED'
GROUP BY DAYNAME(transaction_date), DAYOFWEEK(transaction_date)
ORDER BY DAYOFWEEK(transaction_date);
```
