## 16. SAMPLE SYNTHESIS QUERIES

```sql
-- Month-over-month growth comparison
WITH oct_metrics AS (
  SELECT
    COUNT(DISTINCT user_id) as mau,
    SUM(revenue) as total_revenue,
    COUNT(*) as total_transactions
  FROM user_activity
  WHERE activity_date BETWEEN '2025-10-01' AND '2025-10-31'
),
sept_metrics AS (
  SELECT
    COUNT(DISTINCT user_id) as mau,
    SUM(revenue) as total_revenue,
    COUNT(*) as total_transactions
  FROM user_activity
  WHERE activity_date BETWEEN '2025-09-01' AND '2025-09-30'
)
SELECT
  o.mau as oct_mau,
  s.mau as sept_mau,
  ROUND(((o.mau - s.mau) * 100.0 / s.mau), 2) as mau_growth_pct,
  o.total_revenue as oct_revenue,
  s.total_revenue as sept_revenue,
  ROUND(((o.total_revenue - s.total_revenue) * 100.0 / s.total_revenue), 2) as revenue_growth_pct,
  o.total_transactions as oct_tx,
  s.total_transactions as sept_tx,
  ROUND(((o.total_transactions - s.total_transactions) * 100.0 / s.total_transactions), 2) as tx_growth_pct
FROM oct_metrics o, sept_metrics s;

-- Cohort retention analysis
SELECT
  DATE_FORMAT(signup_date, '%Y-Q%q') as signup_cohort,
  COUNT(*) as cohort_size,
  COUNT(CASE WHEN last_activity_date >= '2025-10-01' THEN 1 END) as active_in_oct,
  ROUND(COUNT(CASE WHEN last_activity_date >= '2025-10-01' THEN 1 END) * 100.0 / COUNT(*), 2) as retention_rate_pct,
  AVG(CASE WHEN last_activity_date >= '2025-10-01' THEN product_count END) as avg_products_per_active_user
FROM users
GROUP BY DATE_FORMAT(signup_date, '%Y-Q%q')
ORDER BY signup_date DESC;

-- Power user analysis (top 10% by volume)
WITH user_volumes AS (
  SELECT
    user_id,
    SUM(amount) as total_volume
  FROM transactions
  WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
    AND status = 'COMPLETED'
  GROUP BY user_id
),
percentiles AS (
  SELECT PERCENTILE_CONT(0.9) WITHIN GROUP (ORDER BY total_volume) as p90
  FROM user_volumes
)
SELECT
  COUNT(*) as power_user_count,
  SUM(v.total_volume) as power_user_volume,
  ROUND(SUM(v.total_volume) * 100.0 / (SELECT SUM(total_volume) FROM user_volumes), 2) as pct_of_total_volume,
  AVG(v.total_volume) as avg_volume_per_power_user
FROM user_volumes v, percentiles p
WHERE v.total_volume >= p.p90;
```
