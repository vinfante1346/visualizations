## 13. SAMPLE SQL QUERIES

```sql
-- Total interchange revenue estimate
SELECT
  SUM(amount) as total_card_spend,
  SUM(amount) * 0.018 as estimated_interchange_revenue,  -- Assuming 1.8% rate
  COUNT(DISTINCT card_id) as active_cards,
  (SUM(amount) * 0.018) / COUNT(DISTINCT card_id) as revenue_per_card
FROM card_transactions
WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'APPROVED';

-- Revenue per user (RPU)
WITH revenue AS (
  SELECT SUM(revenue_amount) as total_revenue
  FROM revenue_events
  WHERE event_date BETWEEN '2025-10-01' AND '2025-10-31'
),
active_users AS (
  SELECT COUNT(DISTINCT user_id) as mau
  FROM user_activity
  WHERE activity_date BETWEEN '2025-10-01' AND '2025-10-31'
)
SELECT
  r.total_revenue,
  a.mau,
  ROUND(r.total_revenue / a.mau, 2) as revenue_per_user
FROM revenue r, active_users a;

-- Cost per transaction
SELECT
  transaction_type,
  COUNT(*) as tx_count,
  SUM(processing_cost) as total_cost,
  AVG(processing_cost) as cost_per_tx,
  AVG(revenue) as revenue_per_tx,
  AVG(revenue - processing_cost) as net_margin_per_tx
FROM transactions
WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
  AND status = 'COMPLETED'
GROUP BY transaction_type
ORDER BY tx_count DESC;

-- Customer acquisition cost
WITH marketing_spend AS (
  SELECT SUM(amount) as total_spend
  FROM marketing_expenses
  WHERE expense_date BETWEEN '2025-10-01' AND '2025-10-31'
),
new_users AS (
  SELECT COUNT(*) as new_user_count
  FROM users
  WHERE created_at BETWEEN '2025-10-01' AND '2025-10-31'
)
SELECT
  m.total_spend,
  n.new_user_count,
  ROUND(m.total_spend / n.new_user_count, 2) as cac
FROM marketing_spend m, new_users n;

-- Profitability by product line
SELECT
  product_line,
  SUM(revenue) as total_revenue,
  SUM(direct_cost) as total_cost,
  SUM(revenue - direct_cost) as gross_profit,
  ROUND((SUM(revenue - direct_cost) / SUM(revenue)) * 100, 2) as gross_margin_pct
FROM product_revenue
WHERE transaction_date BETWEEN '2025-10-01' AND '2025-10-31'
GROUP BY product_line
ORDER BY total_revenue DESC;
```
